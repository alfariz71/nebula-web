// ─── PARTICLE CLASS ──────────────────────────────
class Particle {
    constructor() { this.reset(); }

    reset() {
        this.x       = Math.random() * width;
        this.y       = Math.random() * height;
        this.size    = Math.random() * 2 + 0.5;
        this.speedX  = (Math.random() - 0.5) * 0.5;
        this.speedY  = (Math.random() - 0.5) * 0.5;
        this.alpha   = Math.random() * 0.5 + 0.2;
        this.color   = `rgba(255,${150 + Math.random()*105},${200 + Math.random()*55},${this.alpha})`;
    }

    update() {
        if (isFreePlay) {
            if (currentEffect === 'blackhole') {
                const dx = cursorX - this.x;
                const dy = cursorY - this.y;
                const dist = Math.sqrt(dx*dx + dy*dy);
                if (dist < 400) {
                    this.x += dx * 0.05;
                    this.y += dy * 0.05;
                }
            } else if (currentEffect === 'trail') {
                const dx = cursorX - this.x;
                const dy = cursorY - this.y;
                const dist = Math.sqrt(dx*dx + dy*dy);
                if (dist < 150) {
                    this.speedX = (Math.random() - 0.5) * 8;
                    this.speedY = (Math.random() - 0.5) * 8;
                }
            } else if (currentEffect === 'galaxy') {
                const angle = 0.02;
                const cos = Math.cos(angle), sin = Math.sin(angle);
                const dx = this.x - width/2, dy = this.y - height/2;
                this.x = width/2 + (dx*cos - dy*sin);
                this.y = height/2 + (dx*sin + dy*cos);
            } else if (currentEffect === 'name') {
                // particles drift slowly toward center while name particles form
                const dx = width/2  - this.x;
                const dy = height/2 - this.y;
                this.x += dx * 0.003;
                this.y += dy * 0.003;
            }
        }
        this.x += this.speedX;
        this.y += this.speedY;
        if (this.x < 0 || this.x > width || this.y < 0 || this.y > height) this.reset();
    }

    draw() {
        ctx.shadowBlur  = 10;
        ctx.shadowColor = this.color;
        ctx.beginPath();
        ctx.arc(this.x, this.y, this.size, 0, Math.PI * 2);
        ctx.fillStyle = this.color;
        ctx.fill();
        ctx.shadowBlur = 0;
    }
}

function initParticles() {
    particles = [];
    const count = Math.floor((width * height) / 10000);
    for (let i = 0; i < count; i++) particles.push(new Particle());
}

// ─── MAIN ANIMATE LOOP ───────────────────────────
function animate() {
    ctx.fillStyle = 'rgba(5,5,5,0.2)';
    ctx.fillRect(0, 0, width, height);

    // Background star twinkle
    ctx.fillStyle = 'white';
    for (let i = 0; i < 20; i++) {
        ctx.beginPath();
        ctx.arc(Math.random()*width, Math.random()*height, Math.random()*1.5, 0, Math.PI*2);
        ctx.fill();
    }

    particles.forEach(p => { p.update(); p.draw(); });

    // Trail canvas clear (trail draws on separate canvas)
    if (currentEffect !== 'trail' && currentEffect !== 'blackhole') {
        tctx.clearRect(0, 0, width, height);
    }

    requestAnimationFrame(animate);
}

initParticles();
animate();

//  NAME PARTICLE SYSTEM

function buildNameParticles() {
    nameParticles      = [];
    nameParticlesReady = false;
    nameTransitionStart = performance.now();

    const W = width, H = height;
    const offscreen = document.createElement('canvas');
    offscreen.width  = W;
    offscreen.height = 180;
    const octx = offscreen.getContext('2d');

    // Auto-scale font agar tidak terpotong
    let fontSize = Math.floor(W / 14);
    octx.font = `300 ${fontSize}px 'Segoe UI', Arial, sans-serif`;
    let tw = octx.measureText(NAME_TEXT).width;
    while (tw > W * 0.86 && fontSize > 12) {
        fontSize--;
        octx.font = `300 ${fontSize}px 'Segoe UI', Arial, sans-serif`;
        tw = octx.measureText(NAME_TEXT).width;
    }

    octx.fillStyle      = '#fff';
    octx.textAlign      = 'center';
    octx.textBaseline   = 'middle';
    octx.fillText(NAME_TEXT, W / 2, 90);

    const imgData = octx.getImageData(0, 0, W, 180).data;
    const targetCY = H / 2;

    for (let py = 0; py < 180; py += 3) {
        for (let px = 0; px < W; px += 3) {
            if (imgData[(py * W + px) * 4 + 3] > 100) {
                // Core particle tepat di titik huruf
                nameParticles.push(_makeNameParticle(px, targetCY - 90 + py, W, H, 'core'));

                // Extra sparkle di sekitar huruf (~35% chance)
                if (Math.random() < 0.35) {
                    const a = Math.random() * Math.PI * 2;
                    const r = Math.random() * 9;
                    nameParticles.push(_makeNameParticle(
                        px + Math.cos(a)*r,
                        targetCY - 90 + py + Math.sin(a)*r,
                        W, H, 'sparkle'
                    ));
                }
            }
        }
    }

    nameParticlesReady = true;
}

function _makeNameParticle(tx, ty, W, H, type) {
    // Spawn dari posisi nebula acak di seluruh layar
    const a = Math.random() * Math.PI * 2;
    const r = Math.min(W, H) * (0.25 + Math.random() * 0.75);
    return {
        x:  W/2 + Math.cos(a) * r,
        y:  H/2 + Math.sin(a) * r,
        tx, ty,
        delay:         Math.random() * 1200,   // max 1.2s stagger
        size:          type === 'sparkle' ? Math.random()*1.2+0.4 : Math.random()*1.8+0.8,
        hue:           295 + Math.random() * 65,  // magenta → pink
        sat:           70  + Math.random() * 30,
        lit:           70  + Math.random() * 25,
        alpha:         0,
        isSparkle:     type === 'sparkle',
        twinkleOffset: Math.random() * Math.PI * 2,
    };
}

function drawNameParticles() {
    if (!nameParticlesReady) return;
    const now      = performance.now();
    const DURATION = 3000;
    const elapsed  = now - nameTransitionStart;

    nameParticles.forEach(p => {
        // Per-particle progress 0→1
        const raw  = Math.max(0, Math.min(1, (elapsed - p.delay) / DURATION));
        const ease = 1 - Math.pow(1 - raw, 3);   // ease-out cubic

        // Lerp toward target
        p.x += (p.tx - p.x) * ease * 0.1;
        p.y += (p.ty - p.y) * ease * 0.1;

        p.alpha = Math.min(1, ease * 1.5);

        // Twinkle setelah hampir sampai
        const twinkle = raw > 0.75
            ? 0.72 + Math.sin(now * 0.005 + p.twinkleOffset) * 0.28
            : 1;

        ctx.save();
        ctx.globalAlpha = p.alpha * twinkle;

        if (p.isSparkle) {
            // Bintang silang kecil
            const sz = p.size * (0.8 + twinkle * 0.5);
            ctx.strokeStyle = `hsl(${p.hue},${p.sat}%,${p.lit}%)`;
            ctx.lineWidth   = 0.9;
            ctx.shadowBlur  = 7;
            ctx.shadowColor = `hsl(${p.hue},100%,82%)`;
            ctx.beginPath();
            ctx.moveTo(p.x - sz*2.2, p.y); ctx.lineTo(p.x + sz*2.2, p.y);
            ctx.moveTo(p.x, p.y - sz*2.2); ctx.lineTo(p.x, p.y + sz*2.2);
            ctx.stroke();
        } else {
            // Titik glowing
            ctx.shadowBlur  = 9;
            ctx.shadowColor = `hsl(${p.hue},100%,82%)`;
            ctx.fillStyle   = `hsl(${p.hue},${p.sat}%,${p.lit}%)`;
            ctx.beginPath();
            ctx.arc(p.x, p.y, p.size, 0, Math.PI*2);
            ctx.fill();
        }
        ctx.restore();
    });
}

// ─── TRAIL DRAW ──────────────────────────────────
function drawTrail() {
    tctx.fillStyle = 'rgba(5,5,5,0.18)';
    tctx.fillRect(0, 0, width, height);

    trailPoints = trailPoints.filter(p => p.age < p.life);
    trailPoints.forEach(p => {
        const a = 1 - p.age / p.life;
        if (p.type === 'main') {
            const g = tctx.createRadialGradient(p.x, p.y, 0, p.x, p.y, p.size * 4);
            g.addColorStop(0, `rgba(255,200,255,${a*0.95})`);
            g.addColorStop(0.5, `rgba(255,100,199,${a*0.6})`);
            g.addColorStop(1, `rgba(160,80,240,0)`);
            tctx.beginPath();
            tctx.arc(p.x, p.y, p.size*4, 0, Math.PI*2);
            tctx.fillStyle = g; tctx.fill();
            tctx.beginPath();
            tctx.arc(p.x, p.y, p.size*0.6, 0, Math.PI*2);
            tctx.fillStyle = `rgba(255,255,255,${a})`;
            tctx.fill();
        } else {
            tctx.beginPath();
            tctx.arc(p.x, p.y, p.size, 0, Math.PI*2);
            tctx.fillStyle = `hsla(${290+Math.random()*80},100%,80%,${a*0.8})`;
            tctx.fill();
        }
        p.x += p.vx || 0;
        p.y += (p.vy || 0) + 0.25;
        p.age++;
    });
}

function spawnTrail(x, y) {
    const prev = trailPoints[trailPoints.length - 1];
    const vx   = prev ? (x - prev.x) * 0.25 : 0;
    const vy   = prev ? (y - prev.y) * 0.25 : 0;
    trailPoints.push({ x, y, age:0, life:110, size:Math.random()*3+2.5, vx, vy, type:'main' });
    for (let i = 0; i < 8; i++) {
        trailPoints.push({
            x: x + (Math.random()-0.5)*22,
            y: y + (Math.random()-0.5)*22,
            vx: (Math.random()-0.5)*3,
            vy: (Math.random()-0.5)*3,
            age:0, life:55+~~(Math.random()*50),
            size:Math.random()*1.5+0.4, type:'spark'
        });
    }
    if (trailPoints.length > 900) trailPoints.splice(0, trailPoints.length - 900);
}

// ─── BLACKHOLE DRAW ──────────────────────────────
function drawBlackhole() {
    tctx.fillStyle = 'rgba(5,5,5,0.25)';
    tctx.fillRect(0, 0, width, height);

    const bx = cursorX, by = cursorY;
    const t  = Date.now() / 1000;

    // Dark vortex center
    const vg = tctx.createRadialGradient(bx, by, 0, bx, by, Math.min(width, height)*0.42);
    vg.addColorStop(0,   'rgba(0,0,0,0.88)');
    vg.addColorStop(0.35,'rgba(3,1,10,0.55)');
    vg.addColorStop(1,   'rgba(3,1,10,0)');
    tctx.beginPath();
    tctx.arc(bx, by, Math.min(width, height)*0.42, 0, Math.PI*2);
    tctx.fillStyle = vg; tctx.fill();

    // Multi-ring swirl
    [55, 95, 138, 185].forEach((baseR, ri) => {
        const nPts = 18 + ri*8;
        for (let i = 0; i < nPts; i++) {
            const angle = t*(2.6 - ri*0.4) + i*(Math.PI*2/nPts);
            const r  = baseR + Math.sin(t*2 + i*0.3)*13;
            const px = bx + Math.cos(angle)*r;
            const py = by + Math.sin(angle)*r*0.45;
            const a  = (0.82 - ri*0.13) + Math.sin(t*5+i)*0.18;
            const sz = 3.5 - ri*0.5;
            const pg = tctx.createRadialGradient(px, py, 0, px, py, sz*2.2);
            pg.addColorStop(0, `rgba(255,180,255,${a})`);
            pg.addColorStop(0.5,`rgba(255,70,180,${a*0.7})`);
            pg.addColorStop(1, `rgba(100,0,200,0)`);
            tctx.beginPath();
            tctx.arc(px, py, sz, 0, Math.PI*2);
            tctx.fillStyle = pg; tctx.fill();
        }
    });

    // Inward streaks
    for (let i = 0; i < 18; i++) {
        const ang = t*1.8 + i*(Math.PI*2/18);
        tctx.beginPath();
        tctx.moveTo(bx + Math.cos(ang)*210, by + Math.sin(ang)*210*0.45);
        tctx.lineTo(bx + Math.cos(ang)*38,  by + Math.sin(ang)*38*0.45);
        tctx.strokeStyle = `rgba(200,80,255,${0.07 + Math.sin(t*3+i)*0.04})`;
        tctx.lineWidth = 1.5; tctx.stroke();
    }

    // Singularity glow
    const sg = tctx.createRadialGradient(bx, by, 0, bx, by, 34);
    sg.addColorStop(0,   'rgba(255,255,255,0.95)');
    sg.addColorStop(0.3, 'rgba(255,110,199,0.65)');
    sg.addColorStop(1,   'rgba(0,0,0,0)');
    tctx.beginPath();
    tctx.arc(bx, by, 34, 0, Math.PI*2);
    tctx.fillStyle = sg; tctx.fill();
}

// ─── FREEPLAY DRAW LOOP ──────────────────────────
function freeplayLoop() {

    if (currentEffect === 'trail')     drawTrail();
    if (currentEffect === 'blackhole') drawBlackhole();
    if (currentEffect === 'name')      drawNameParticles();
    requestAnimationFrame(freeplayLoop);
}

// ─── QUESTION FLOW ───────────────────────────────
function showNextQuestion() {
    if (STATE.currentQuestionIndex >= STATE.questions.length) {
        finishQuestions();
        return;
    }
    const question = STATE.questions[STATE.currentQuestionIndex];
    questionContainer.classList.remove('hidden');
    questionContainer.style.opacity = '0';

    setTimeout(() => {
        questionText.textContent  = question.text;
        optionsContainer.innerHTML = '';
        question.options.forEach((option, index) => {
            const btn = document.createElement('button');
            btn.className    = 'option-btn';
            btn.textContent  = option;
            btn.dataset.index = index;
            btn.addEventListener('click', () => selectOption(index));
            optionsContainer.appendChild(btn);
        });
        questionContainer.style.opacity = '1';
    }, 500);
}

function selectOption(index) {

    btns[index].classList.add('selected');
    STATE.answers.push(STATE.questions[STATE.currentQuestionIndex].options[index]);

    setTimeout(() => {
        questionContainer.style.opacity = '0';
        setTimeout(() => {
            STATE.currentQuestionIndex++;
            showNextQuestion();
        }, 800);
    }, 600);
}

function captureAndStartPuzzle() {
    const pctx = captureCanvas.getContext('2d');
    captureCanvas.width  = 600;
    captureCanvas.height = 400;
    pctx.translate(600, 0);
    pctx.scale(-1, 1);
    pctx.drawImage(videoElement, 0, 0, 600, 400);
    const imageData = captureCanvas.toDataURL('image/jpeg');
    createPuzzlePieces(imageData);
    startTime    = Date.now();
    isGameActive = true;
}

function createPuzzlePieces(imageSrc) {
    puzzleBoard.innerHTML = '';
    puzzlePieces = [];
    const rows = 2, cols = 3;
    const pw = 600/cols, ph = 400/rows;

    for (let r = 0; r < rows; r++) {
        for (let c = 0; c < cols; c++) {
            const piece = document.createElement('div');
            piece.className = 'puzzle-piece';
            piece.style.width  = `${pw}px`;
            piece.style.height = `${ph}px`;
            piece.style.backgroundImage    = `url(${imageSrc})`;
            piece.style.backgroundPosition = `-${c*pw}px -${r*ph}px`;
            const pObj = {
                element:  piece,
                targetX:  c * pw,
                targetY:  r * ph,
                currentX: Math.random() * (window.innerWidth  - pw),
                currentY: Math.random() * (window.innerHeight - ph),
                id: r*cols + c
            };
            updatePiecePosition(pObj);
            puzzleBoard.appendChild(piece);
            puzzlePieces.push(pObj);
        }
    }
    scramblePieces();
}

function updatePiecePosition(p) {
    p.element.style.left = `${p.currentX}px`;
    p.element.style.top  = `${p.currentY}px`;
}

function scramblePieces() {
    puzzlePieces.forEach(p => {
        p.currentX = Math.random() * 400;
        p.currentY = Math.random() * 300;
        updatePiecePosition(p);
    });
}

function checkSnap(piece) {
    const dx = piece.targetX - piece.currentX;
    const dy = piece.targetY - piece.currentY;
    if (Math.sqrt(dx*dx + dy*dy) < 60) {
        piece.currentX = piece.targetX;
        piece.currentY = piece.targetY;
        updatePiecePosition(piece);
        piece.element.classList.add('correct');
        piece.locked = true;
        piece.element.style.transition  = '0.2s ease';
        piece.element.style.boxShadow   = '0 0 20px rgba(255,182,193,0.8)';
        setTimeout(() => { piece.element.style.boxShadow = ''; }, 250);
    }
}

function checkWin() {
    const allCorrect = puzzlePieces.every(p => p.element.classList.contains('correct'));
    if (allCorrect && isGameActive) {
        isGameActive = false;
        finishPuzzle();
    }
}

let hands, camera;

function startCamera() {
    hands = new Hands({
        locateFile: file => `https://cdn.jsdelivr.net/npm/@mediapipe/hands/${file}`
    });
    hands.setOptions({
        maxNumHands: 1,
        modelComplexity: 1,
        minDetectionConfidence: 0.7,
        minTrackingConfidence:  0.7
    });
    hands.onResults(onResults);
    camera = new Camera(videoElement, {
        onFrame: async () => { await hands.send({ image: videoElement }); },
        width: 640, height: 480
    });
    camera.start();
}

const lerp = (s, e, a) => (1-a)*s + a*e;

function onResults(results) {
    if (!results.multiHandLandmarks || results.multiHandLandmarks.length === 0) return;
    const lm = results.multiHandLandmarks[0];
    targetX = (1 - lm[8].x) * window.innerWidth;
    targetY = lm[8].y * window.innerHeight;
    detectGestures(lm);
}

// ─── GESTURE DETECTION ───────────────────────────
// NOTE: currentEffect declared at top — no hoisting issue
function detectGestures(lm) {
    const thumbTip  = lm[4];
    const indexTip  = lm[8];
    const middleTip = lm[12];
    const ringTip   = lm[16];
    const pinkyTip  = lm[20];

    const indexExt  = indexTip.y  < lm[6].y;
    const middleExt = middleTip.y < lm[10].y;
    const ringExt   = ringTip.y   < lm[14].y;
    const pinkyExt  = pinkyTip.y  < lm[18].y;

    // ── FREEPLAY GESTURES ──
    if (isFreePlay) {
        let newEffect = 'none';
        if      ( indexExt &&  middleExt && !ringExt && !pinkyExt)  newEffect = 'name';      // ✌️
        else if ( indexExt &&  middleExt &&  ringExt &&  pinkyExt)  newEffect = 'galaxy';    // 🖐️
        else if ( indexExt && !middleExt && !ringExt && !pinkyExt)  newEffect = 'trail';     // ☝️
        else if (!indexExt && !middleExt && !ringExt && !pinkyExt)  newEffect = 'blackhole'; // ✊

        if (newEffect !== currentEffect) {
            currentEffect = newEffect;
            // Rebuild name particles every time ✌️ is activated
            if (currentEffect === 'name') {
                tctx.clearRect(0, 0, width, height);
                buildNameParticles();
            }
            // Clear trail canvas when switching away
            if (currentEffect !== 'trail' && currentEffect !== 'blackhole') {
                tctx.clearRect(0, 0, width, height);
            }
        }

        // Spawn trail points while ☝️ is held
        if (currentEffect === 'trail') spawnTrail(cursorX, cursorY);
    }

    // ── PINCH ──
    const pinchDist = Math.hypot(indexTip.x - thumbTip.x, indexTip.y - thumbTip.y);
    if (pinchDist < 0.04) {
        if (!isPinching) {
            isPinching = true;
            cursor.classList.add('clicking');
            triggerGesture('click');
        }
    } else {
        if (isPinching) {
            isPinching = false;
            cursor.classList.remove('clicking');
        }
    }

    // ── BACK GESTURE ──
    const isMiddleCurled  = middleTip.y > lm[10].y;
    const isPointingLeft  = indexTip.x  > lm[5].x + 0.1;
    if (indexExt && isMiddleCurled && isPointingLeft) debounceGesture('back');

    // ── PUZZLE DRAG ──
    if (isPinching && draggedPiece) {
        const mx = (targetX - cursorX) * 1.8;
        const my = (targetY - cursorY) * 1.8;
        draggedPiece.currentX += mx * 0.2;
        draggedPiece.currentY += my * 0.2;
        updatePiecePosition(draggedPiece);
    } else if (!isPinching && draggedPiece) {
        checkSnap(draggedPiece);
        draggedPiece.element.classList.remove('dragging');
        draggedPiece = null;
        checkWin();
    }
}

function debounceGesture(gesture) {
    if (lastGesture === gesture) return;
    if (gestureDebounceTimer) clearTimeout(gestureDebounceTimer);
    gestureDebounceTimer = setTimeout(() => {
        triggerGesture(gesture);
        lastGesture = gesture;
        setTimeout(() => { lastGesture = null; }, 1000);
    }, 150);
}

function triggerGesture(type) {
    if (type === 'click') {
        if (isStoryActive) { nextStoryLine(); return; }
        const hovered = document.querySelector(
            '.option-btn.hovered, #next-to-puzzle.hovered, #start-puzzle-btn.hovered, ' +
            '#next-to-story.hovered, #replay-btn.hovered, #start-freeplay-btn.hovered, ' +
            '#opening-start-btn.hovered'
        );
        if (hovered) hovered.click();

        if (isGameActive && !draggedPiece) {
            const els    = document.elementsFromPoint(cursorX, cursorY);
            const pieceEl = els.find(el => el.classList.contains('puzzle-piece'));
            if (pieceEl) {
                draggedPiece = puzzlePieces.find(p => p.element === pieceEl);
                if (draggedPiece && draggedPiece.locked) { draggedPiece = null; return; }
                if (draggedPiece) draggedPiece.element.classList.add('dragging');
            }
        }
    }
    if (type === 'back') {
        cursor.style.boxShadow = '0 0 30px rgba(255,255,255,0.8)';
        setTimeout(() => { cursor.style.boxShadow = '0 0 15px rgba(255,105,180,0.8)'; }, 500);
    }
}

// ─── CURSOR UPDATE LOOP ──────────────────────────
function updateCursor() {
    cursorX = lerp(cursorX, targetX, 0.15);
    cursorY = lerp(cursorY, targetY, 0.15);
    cursor.style.left = `${cursorX}px`;
    cursor.style.top  = `${cursorY}px`;

    const els = document.elementsFromPoint(cursorX, cursorY);
    const hit = els.find(el =>
        el.classList.contains('option-btn') ||
        el.id === 'next-to-puzzle'   ||
        el.id === 'start-puzzle-btn' ||
        el.id === 'next-to-story'    ||
        el.id === 'opening-start-btn'||
        el.classList.contains('puzzle-piece')
    );
    document.querySelectorAll(
        '.option-btn, #next-to-puzzle, #start-puzzle-btn, #next-to-story, ' +
        '#replay-btn, #start-freeplay-btn, #opening-start-btn'
    ).forEach(btn => btn.classList.toggle('hovered', btn === hit));

    requestAnimationFrame(updateCursor);
}
