class Particle {
    constructor() {
        this.reset();
    }

    reset() {
        this.x = Math.random() * width;
        this.y = Math.random() * height;
        this.targetX = this.x;
        this.targetY = this.y;
        this.size = Math.random() * 2 + 0.5;
        this.speedX = (Math.random() - 0.5) * 0.5;
        this.speedY = (Math.random() - 0.5) * 0.5;
        this.alpha = Math.random() * 0.5 + 0.2;
        this.color = `rgba(255, ${150 + Math.random() * 105}, ${200 + Math.random() * 55}, ${this.alpha})`;
        this.mode = 'float';
    }

    update() {
        if (isFreePlay) {
            if (currentEffect === 'blackhole') {
                const dx = cursorX - this.x;
                const dy = cursorY - this.y;
                const dist = Math.sqrt(dx * dx + dy * dy);
                if (dist < 400) {
                    this.x += dx * 0.05;
                    this.y += dy * 0.05;
                }
            } else if (currentEffect === 'trail') {
                const dx = cursorX - this.x;
                const dy = cursorY - this.y;
                const dist = Math.sqrt(dx * dx + dy * dy);
                if (dist < 150) {
                    this.speedX = (Math.random() - 0.5) * 8;
                    this.speedY = (Math.random() - 0.5) * 8;
                }
            } else if (currentEffect === 'galaxy') {
                const angle = 0.02;
                const cos = Math.cos(angle);
                const sin = Math.sin(angle);
                const dx = this.x - width/2;
                const dy = this.y - height/2;
                this.x = width/2 + (dx * cos - dy * sin);
                this.y = height/2 + (dx * sin + dy * cos);
            } else if (currentEffect === 'name') {
                // Name mode: drift towards center text area
                const dx = width/2 - this.x;
                const dy = height/2 - this.y;
                this.x += dx * 0.01;
                this.y += dy * 0.01;
            }
        }

        this.x += this.speedX;
        this.y += this.speedY;

        if (this.x < 0 || this.x > width || this.y < 0 || this.y > height) {
            this.reset();
        }
    }

    draw() {
        ctx.beginPath();
        ctx.arc(this.x, this.y, this.size, 0, Math.PI * 2);
        ctx.fillStyle = this.color;
        ctx.fill();
        
        // Add soft glow
        ctx.shadowBlur = 10;
        ctx.shadowColor = this.color;
    }
}

function initParticles() {
    particles = [];
    const count = Math.floor((width * height) / 10000); // Responsive particle count
    for (let i = 0; i < count; i++) {
        particles.push(new Particle());
    }
}

function animate() {
    // Semi-transparent clear to create trail effect
    ctx.fillStyle = 'rgba(5, 5, 5, 0.2)';
    ctx.fillRect(0, 0, width, height);
    
    // Draw background stars
    ctx.fillStyle = 'white';
    for(let i = 0; i < 20; i++) {
        const x = Math.random() * width;
        const y = Math.random() * height;
        const s = Math.random() * 1.5;
        ctx.beginPath();
        ctx.arc(x, y, s, 0, Math.PI * 2);
        ctx.fill();
    }

    particles.forEach(p => {
        p.update();
        p.draw();
    });

    if (isFreePlay && currentEffect === 'name') {
        ctx.fillStyle = 'rgba(255, 105, 180, 0.8)';
        ctx.font = '30px Arial';
        ctx.textAlign = 'center';
        ctx.fillText("Meisya Qeena Sheehan Rusmiyanto", width/2, height/2);
    }

    requestAnimationFrame(animate);
}

function captureAndStartPuzzle() {
    const ctx = captureCanvas.getContext('2d');
    captureCanvas.width = 600;
    captureCanvas.height = 400;
    
    // Draw mirrored webcam frame to canvas
    ctx.translate(600, 0);
    ctx.scale(-1, 1);
    ctx.drawImage(videoElement, 0, 0, 600, 400);
    
    const imageData = captureCanvas.toDataURL('image/jpeg');
    createPuzzlePieces(imageData);
    startTime = Date.now();
    isGameActive = true;
}

function createPuzzlePieces(imageSrc) {
    puzzleBoard.innerHTML = '';
    puzzlePieces = [];
    
    const rows = 2;
    const cols = 3;
    const pieceWidth = 600 / cols;
    const pieceHeight = 400 / rows;

    for (let r = 0; r < rows; r++) {
        for (let c = 0; c < cols; c++) {
            const piece = document.createElement('div');
            piece.className = 'puzzle-piece';
            piece.style.width = `${pieceWidth}px`;
            piece.style.height = `${pieceHeight}px`;
            piece.style.backgroundImage = `url(${imageSrc})`;
            piece.style.backgroundPosition = `-${c * pieceWidth}px -${r * pieceHeight}px`;
            
            // Initial random position
            const startX = Math.random() * (window.innerWidth - pieceWidth);
            const startY = Math.random() * (window.innerHeight - pieceHeight);
            
            const targetX = c * pieceWidth;
            const targetY = r * pieceHeight;

            const pieceObj = {
                element: piece,
                targetX: targetX,
                targetY: targetY,
                currentX: startX,
                currentY: startY,
                id: r * cols + c
            };

            updatePiecePosition(pieceObj);
            puzzleBoard.appendChild(piece);
            puzzlePieces.push(pieceObj);
        }
    }
    
    // Scramble pieces
    scramblePieces();
}

function updatePiecePosition(pieceObj) {
    pieceObj.element.style.left = `${pieceObj.currentX}px`;
    pieceObj.element.style.top = `${pieceObj.currentY}px`;
}

function scramblePieces() {
    const boardRect = puzzleBoard.getBoundingClientRect();
    puzzlePieces.forEach(p => {
        // Scatter around the board but not exactly on target
        p.currentX = Math.random() * 400;
        p.currentY = Math.random() * 300;
        updatePiecePosition(p);
    });
}

let hands;
let camera;

function startCamera() {
    hands = new Hands({
        locateFile: (file) => {
            return `https://cdn.jsdelivr.net/npm/@mediapipe/hands/${file}`;
        }
    });

    hands.setOptions({
        maxNumHands: 1,
        modelComplexity: 1,
        minDetectionConfidence: 0.7,
        minTrackingConfidence: 0.7
    });

    hands.onResults(onResults);

    camera = new Camera(videoElement, {
        onFrame: async () => {
            await hands.send({ image: videoElement });
        },
        width: 640,
        height: 480
    });
    camera.start();
}

let cursorX = 0;
let cursorY = 0;
let targetX = 0;
let targetY = 0;

// Smoothing factor
const lerp = (start, end, amt) => (1 - amt) * start + amt * end;

function onResults(results) {
    if (results.multiHandLandmarks && results.multiHandLandmarks.length > 0) {
        const landmarks = results.multiHandLandmarks[0];
        
        // Use Index Finger Tip (Landmark 8) for cursor movement
        const indexTip = landmarks[8];
        const thumbTip = landmarks[4];
        const middleTip = landmarks[12];
        
        // Update target position (mirrored)
        targetX = (1 - indexTip.x) * window.innerWidth;
        targetY = indexTip.y * window.innerHeight;

        // Detect Gestures
        detectGestures(landmarks);
    }
}

let isPinching = false;
let lastGesture = null;
let gestureDebounceTimer = null;

function detectGestures(landmarks) {
    const thumbTip = landmarks[4];
    const indexTip = landmarks[8];
    const middleTip = landmarks[12];
    const ringTip = landmarks[16];
    const pinkyTip = landmarks[20];
    
    const indexExtended = indexTip.y < landmarks[6].y;
    const middleExtended = middleTip.y < landmarks[10].y;
    const ringExtended = ringTip.y < landmarks[14].y;
    const pinkyExtended = pinkyTip.y < landmarks[18].y;

    if (isFreePlay) {
        // Free Play Gestures
        if (indexExtended && middleExtended && !ringExtended && !pinkyExtended) {
            currentEffect = 'name'; // ✌️
        } else if (indexExtended && middleExtended && ringExtended && pinkyExtended) {
            currentEffect = 'galaxy'; // 🖐️
        } else if (indexExtended && !middleExtended && !ringExtended && !pinkyExtended) {
            currentEffect = 'trail'; // ☝️
        } else if (!indexExtended && !middleExtended && !ringExtended && !pinkyExtended) {
            currentEffect = 'blackhole'; // ✊
        } else {
            currentEffect = 'none';
        }
    }

    const pinchDistance = Math.sqrt(
        Math.pow(indexTip.x - thumbTip.x, 2) +
        Math.pow(indexTip.y - thumbTip.y, 2)
    );

    const isIndexExtended = indexTip.y < landmarks[6].y;
    const isMiddleCurled = middleTip.y > landmarks[10].y;
    const isPointingLeft = indexTip.x > landmarks[5].x + 0.1;

    if (pinchDistance < 0.04) {
        if (!isPinching) {
            isPinching = true;
            cursor.classList.add('clicking');
            triggerGesture('click');
        }
    } else {
        if (isPinching) {
            isPinching = false;
            cursor.classList.remove('clicking');
        }
    }

    if (isIndexExtended && isMiddleCurled && isPointingLeft) {
        debounceGesture('back');
    }

    // Handle Drag Movement
    if (isPinching && draggedPiece) {

    // smooth delta movement
    const moveX = (targetX - cursorX) * 1.8;
    const moveY = (targetY - cursorY) * 1.8;

    draggedPiece.currentX += moveX * 0.2;
    draggedPiece.currentY += moveY * 0.2;

    updatePiecePosition(draggedPiece);
    } else if (!isPinching && draggedPiece) {
        // Drop Piece
        checkSnap(draggedPiece);
        draggedPiece.element.classList.remove('dragging');
        draggedPiece = null;
        checkWin();
    }
}

function checkSnap(piece) {

    const dx = piece.targetX - piece.currentX;
    const dy = piece.targetY - piece.currentY;

    const distance = Math.sqrt(dx * dx + dy * dy);

    const snapRadius = 60;

    if (distance < snapRadius) {

        // SNAP LANGSUNG
        piece.currentX = piece.targetX;
        piece.currentY = piece.targetY;

        updatePiecePosition(piece);

        piece.element.classList.add('correct');
        piece.locked = true;

        piece.element.style.transition = '0.2s ease';
        piece.element.style.boxShadow =
            '0 0 20px rgba(255,182,193,0.8)';

        setTimeout(() => {
            piece.element.style.boxShadow = '';
        }, 250);
    }
}

function checkWin() {
    const allCorrect = puzzlePieces.every(p => p.element.classList.contains('correct'));
    if (allCorrect && isGameActive) {
        isGameActive = false;
        finishPuzzle();
    }
}

    // Move previous lines up

    lines.forEach(line => line.classList.add('scrolled'));
    
    // Create new line
    const lineEl = document.createElement('div');
    lineEl.className = 'story-line';
    lineEl.textContent = lineText;
    storyScroll.appendChild(lineEl);
    
    // Offset the container to keep current line centered
    const offset = currentLineIndex * 60; // Approximate line height + margin
    storyScroll.style.transform = `translateY(-${offset}px)`;

    // Fade in new line
    setTimeout(() => {
        lineEl.classList.add('active');
    }, 100);

    // Auto next if ends with "..."
    if (lineText.endsWith('...')) {
        autoNextTimer = setTimeout(nextStoryLine, 4000);
    }
}

    setTimeout(() => {
        storyContainer.style.opacity = '0';
        storyContainer.style.transition = 'opacity 3s ease';
        setTimeout(showFinalEnding, 3000);
    }, 4000);
}

function debounceGesture(gesture) {
    if (lastGesture === gesture) return;
    
    if (gestureDebounceTimer) clearTimeout(gestureDebounceTimer);
    
    gestureDebounceTimer = setTimeout(() => {
        triggerGesture(gesture);
        lastGesture = gesture;
        // Reset last gesture after a while to allow repeated triggers
        setTimeout(() => { lastGesture = null; }, 1000);
    }, 150);
}

function triggerGesture(type) {
    console.log(`Gesture Triggered: ${type}`);
    
    if (type === 'click') {
        // Handle Storytelling interaction
        if (isStoryActive) {
            nextStoryLine();
            return;
        }
        // Find if cursor is over any option button or next button
        const hoveredBtn = document.querySelector('.option-btn.hovered, #next-to-puzzle.hovered, #start-puzzle-btn.hovered, #next-to-story.hovered, #replay-btn.hovered, #start-freeplay-btn.hovered');
        if (hoveredBtn) {
            hoveredBtn.click();
        }

        // Handle Puzzle Drag Start
        if (isGameActive && !draggedPiece) {
            const elements = document.elementsFromPoint(cursorX, cursorY);
            const pieceEl = elements.find(el => el.classList.contains('puzzle-piece'));
            if (pieceEl) {

                draggedPiece = puzzlePieces.find(p => p.element === pieceEl);

                // JANGAN BISA DIGERAKKAN LAGI
                if (draggedPiece.locked) return;

                draggedPiece.element.classList.add('dragging');

            }
        }
    }

    // Visual feedback for back gesture
    if (type === 'back') {
        cursor.style.boxShadow = '0 0 30px rgba(255, 255, 255, 0.8)';
        setTimeout(() => {
            cursor.style.boxShadow = '0 0 15px rgba(255, 105, 180, 0.8)';
        }, 500);
        
        // Potential back logic if needed
        if (STATE.currentQuestionIndex > 0) {
             // Optional: handle back navigation
        }
    }
}

// Update cursor position smoothly and handle hover detection
function updateCursor() {
    cursorX = lerp(cursorX, targetX, 0.15);
    cursorY = lerp(cursorY, targetY, 0.15);
    
    cursor.style.left = `${cursorX}px`;
    cursor.style.top = `${cursorY}px`;

    // Check for hover on options
    const elements = document.elementsFromPoint(cursorX, cursorY);
    const interactiveEl = elements.find(el => 
        el.classList.contains('option-btn') || 
        el.id === 'next-to-puzzle' || 
        el.id === 'start-puzzle-btn' ||
        el.id === 'next-to-story' ||
        el.classList.contains('puzzle-piece')
    );
    
    document.querySelectorAll('.option-btn, #next-to-puzzle, #start-puzzle-btn, #next-to-story, #replay-btn, #start-freeplay-btn').forEach(btn => {
        if (btn === interactiveEl) {
            btn.classList.add('hovered');
        } else {
            btn.classList.remove('hovered');
        }
    });
    
    requestAnimationFrame(updateCursor);
}
